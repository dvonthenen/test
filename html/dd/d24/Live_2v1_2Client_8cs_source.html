62  public async Task Connect(LiveSchema options, CancellationTokenSource? cancelToken = null, Dictionary<string, string>? addons = null,
63  Dictionary<string, string>? headers = null)
64  {
65  Log.Verbose("LiveClient.Connect", "ENTER");
66  Log.Information("Connect", $"options:\n{JsonSerializer.Serialize(options, JsonSerializeOptions.DefaultOptions)}");
67  Log.Debug("Connect", $"addons: {addons}");
68 
69  // check if the client is disposed
70  if (_clientWebSocket != null)
71  {
72  // client has already connected
73  var exStr = "Client has already been initialized";
74  Log.Error("Connect", exStr);
75  Log.Verbose("LiveClient.Connect", "LEAVE");
76  throw new InvalidOperationException(exStr);
77  }
78 
79  if (cancelToken == null)
80  {
81  Log.Information("Connect", "Using default connect cancellation token");
82  cancelToken = new CancellationTokenSource(Constants.DefaultConnectTimeout);
83  }
84 
85  // create client
86  _clientWebSocket = new ClientWebSocket();
87 
88  // set headers
89  _clientWebSocket.Options.SetRequestHeader("Authorization", $"token {_deepgramClientOptions.ApiKey}");
90  if (_deepgramClientOptions.Headers is not null) {
91  foreach (var header in _deepgramClientOptions.Headers)
92  {
93  var tmp = header.Key.ToLower();
94  if (!(tmp.Contains("password") || tmp.Contains("token") || tmp.Contains("authorization") || tmp.Contains("auth")))
95  {
96  Log.Debug("PutAsync<S, T>", $"Add Header {header.Key}={header.Value}");
97  }
98  _clientWebSocket.Options.SetRequestHeader(header.Key, header.Value);
99  }
100  }
101  if (headers is not null)
102  {
103  foreach (var header in headers)
104  {
105  var tmp = header.Key.ToLower();
106  if (!(tmp.Contains("password") || tmp.Contains("token") || tmp.Contains("authorization") || tmp.Contains("auth")))
107  {
108  Log.Debug("PutAsync<S, T>", $"Add Header {header.Key}={header.Value}");
109  }
110  _clientWebSocket.Options.SetRequestHeader(header.Key, header.Value);
111  }
112  }
113 
114  // internal cancelation token for internal threads
115  _cancellationTokenSource = new CancellationTokenSource();
116 
117  try
118  {
119  var _uri = GetUri(_deepgramClientOptions, options, addons);
120  Log.Debug("Connect", $"uri: {_uri}");
121 
122  Log.Debug("Connect", "Connecting to Deepgram API...");
123  await _clientWebSocket.ConnectAsync(_uri, cancelToken.Token).ConfigureAwait(false);
124 
125  Log.Debug("Connect", "Starting Sender Thread...");
126  StartSenderBackgroundThread();
127 
128  Log.Debug("Connect", "Starting Receiver Thread...");
129  StartReceiverBackgroundThread();
130 
131  if (_deepgramClientOptions.KeepAlive)
132  {
133  Log.Debug("Connect", "Starting KeepAlive Thread...");
134  StartKeepAliveBackgroundThread();
135  }
136 
137  // send a OpenResponse event
138  if (_openReceived != null)
139  {
140  Log.Debug("Connect", "Sending OpenResponse event...");
141  var data = new OpenResponse();
142  data.Type = LiveType.Open;
143  _openReceived.Invoke(null, data);
144  }
145 
146  Log.Debug("Connect", "Connect Succeeded");
147  Log.Verbose("LiveClient.Connect", "LEAVE");
148  }
149  catch (TaskCanceledException ex)
150  {
151  Log.Debug("Connect", "Connect cancelled.");
152  Log.Verbose("Connect", $"Connect cancelled. Info: {ex}");
153  Log.Verbose("LiveClient.Connect", "LEAVE");
154  }
155  catch (Exception ex)
156  {
157  Log.Error("Connect", $"{ex.GetType()} thrown {ex.Message}");
158  Log.Verbose("Connect", $"Excepton: {ex}");
159  Log.Verbose("LiveClient.Connect", "LEAVE");
160  throw;
161  }

343  {

345

440  Log.Verbose("LiveClient.ProcessReceiveQueue", "ENTER");
441 
442  while (_clientWebSocket?.State == WebSocketState.Open)
443  {
444  try
445  {
446  if (_cancellationTokenSource.Token.IsCancellationRequested)
447  {
448  Log.Information("ProcessReceiveQueue", "ReceiveThread cancelled");
449  await Stop();
450  Log.Verbose("ProcessReceiveQueue", "LEAVE");
451  return;
452  }
453 
454  var buffer = new ArraySegment<byte>(new byte[Constants.BufferSize]);
455  WebSocketReceiveResult result;
456 
457  using (var ms = new MemoryStream())
458  {
459  do
460  {
461  // get the result of the receive operation
462  result = await _clientWebSocket.ReceiveAsync(buffer, _cancellationTokenSource.Token);
463 
464  ms.Write(
465  buffer.Array ?? throw new InvalidOperationException("buffer cannot be null"),
466  buffer.Offset,
467  result.Count
468  );
469  } while (!result.EndOfMessage);
470 
471  if (result.MessageType != WebSocketMessageType.Close)
472  {
473  Log.Verbose("ProcessReceiveQueue", $"Received message: {result} / {ms}");
474  ProcessDataReceived(result, ms);
475  }
476  }
477 
478  if (result.MessageType == WebSocketMessageType.Close)
479  {
480  Log.Information("ProcessReceiveQueue", "Received WebSocket Close. Trigger cancel...");
481  await Stop();
482  Log.Verbose("ProcessReceiveQueue", "LEAVE");
483  return;
484  }
485  }
486  catch (TaskCanceledException ex)
487  {
488  Log.Debug("ProcessReceiveQueue", "ReceiveThread cancelled.");
489  Log.Verbose("ProcessReceiveQueue", $"ReceiveThread cancelled. Info: {ex}");

501  internal void ProcessDataReceived(WebSocketReceiveResult result, MemoryStream ms)
502  {
503  Log.Verbose("LiveClient.ProcessDataReceived", "ENTER");
504 
505  ms.Seek(0, SeekOrigin.Begin);
506 
507  if (result.MessageType != WebSocketMessageType.Text)
508  {
509  Log.Warning("ProcessDataReceived", "Received a text message. This is not supported.");
510  Log.Verbose("LiveClient.ProcessDataReceived", "LEAVE");
511  return;
512  }
513 
514  var response = Encoding.UTF8.GetString(ms.ToArray());
515  if (response == null)
516  {
517  Log.Warning("ProcessDataReceived", "Response is null");
518  Log.Verbose("LiveClient.ProcessDataReceived", "LEAVE");
519  return;
520  }
521 
522  try
523  {
524  Log.Verbose("ProcessDataReceived", $"raw response: {response}");
525  var data = JsonDocument.Parse(response);
526  var val = Enum.Parse(typeof(LiveType), data.RootElement.GetProperty("type").GetString()!);
527 
528  Log.Verbose("ProcessDataReceived", $"Type: {val}");
529  switch (val)
530  {
531  case LiveType.Open:
532  var openResponse = data.Deserialize<OpenResponse>();
533  if (_openReceived == null)
534  {
535  Log.Debug("ProcessDataReceived", "_openReceived has no listeners");
536  Log.Verbose("ProcessDataReceived", "LEAVE");
537  return;
538  }
539  if (openResponse == null)
540  {
541  Log.Warning("ProcessDataReceived", "OpenResponse is invalid");
542  Log.Verbose("ProcessDataReceived", "LEAVE");
543  return;
544  }
545 
546  Log.Debug("ProcessDataReceived", $"Invoking OpenResponse. event: {openResponse}");
547  InvokeParallel(_openReceived, openResponse);
548  break;
549  case LiveType.Results:
550  var resultResponse = data.Deserialize<ResultResponse>();
551  if (_resultsReceived == null)
552  {
553  Log.Debug("ProcessDataReceived", "_resultsReceived has no listeners");
554  Log.Verbose("ProcessDataReceived", "LEAVE");
555  return;
556  }

562  }
563 
564  Log.Debug("ProcessDataReceived", $"Invoking ResultsResponse. event: {resultResponse}");

566  break;
567  case LiveType.Metadata:

570  {
571  Log.Debug("ProcessDataReceived", "_metadataReceived has no listeners");

695  public async Task Stop(CancellationTokenSource? cancelToken = null)
696  {
697  Log.Verbose("LiveClient.Stop", "ENTER");
698 
699  // client is already disposed
700  if (_clientWebSocket == null)
701  {
702  Log.Information("Stop", "Client has already been disposed");
703  Log.Verbose("LiveClient.Stop", "LEAVE");
704  return;
705  }
706 
707  if (cancelToken == null)
708  {
709  Log.Information("Stop", "Using default disconnect cancellation token");
710  cancelToken = new CancellationTokenSource(Constants.DefaultDisconnectTimeout);
711  }
712 
713  try
714  {
715  // cancel the internal token to stop all threads
716  if (_cancellationTokenSource != null)
717  {
718  Log.Debug("Stop", "Cancelling native token...");
719  _cancellationTokenSource.Cancel();
720  }
721 
722  // if websocket is open, send a close message
723  if (_clientWebSocket!.State == WebSocketState.Open)
724  {
725  Log.Debug("Stop", "Sending Close message...");
726  // send a close to Deepgram
727  lock (_mutexSend)
728  {
729  _clientWebSocket.SendAsync(new ArraySegment<byte>([0]), WebSocketMessageType.Binary, true, cancelToken.Token)
730  .ConfigureAwait(false);
731  }
732  }
733 
734  // send a CloseResponse event
735  if (_closeReceived != null)
736  {
737  Log.Debug("Stop", "Sending CloseResponse event...");
738  var data = new CloseResponse();
739  data.Type = LiveType.Close;
740  InvokeParallel(_closeReceived, data);
741  }
742 
743  // attempt to stop the connection
744  if (_clientWebSocket!.State != WebSocketState.Closed && _clientWebSocket!.State != WebSocketState.Aborted)
745  {
746  Log.Debug("Stop", "Closing WebSocket connection...");
747  await _clientWebSocket.CloseOutputAsync(WebSocketCloseStatus.NormalClosure, string.Empty, cancelToken.Token)
748  .ConfigureAwait(false);
749  }
750 
751  // clean up internal token
752  if (_cancellationTokenSource != null)
753  {
754  Log.Debug("Stop", "Disposing internal token...");
755  _cancellationTokenSource.Dispose();
756  _cancellationTokenSource = null;
757  }
758 
759  // release the socket
760  Log.Debug("Stop", "Disposing WebSocket socket...");
761  _clientWebSocket = null;
762 
763  Log.Debug("Stop", "Succeeded");
764  Log.Verbose("LiveClient.Stop", "LEAVE");
765  }
766  catch (TaskCanceledException ex)
767  {
768  Log.Debug("Stop", "Stop cancelled.");
769  Log.Verbose("Stop", $"Stop cancelled. Info: {ex}");
770  Log.Verbose("LiveClient.Stop", "LEAVE");
